class Dog {
    constructor(name){
        this.name = name
    }
}
class Cat {
    constructor(name){
        this.name = name
    }
    purrNTimes(n){
        for(let i = 0; i <n ; i++){
            console.log(`{this.name} say: "meow");
        }
    }
}
let cat = new Cat("Daisy");
let dog = new Dog("Buba")
cat.purrNTimes(dog, 2);
cat.purrNTimes(cat, [3]);
// prints
Buba says: meow
Buba says: meow
Daisy says: meow
Daisy says: meow
Daisy says: meow
