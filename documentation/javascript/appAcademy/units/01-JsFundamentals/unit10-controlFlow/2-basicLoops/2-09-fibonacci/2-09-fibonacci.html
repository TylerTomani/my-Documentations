9. fibonacci()

Begins with the number 1, adds 1 to itself and proceeds to add the last sum to the previous number in the sequence

[ 1, 1, 2 ,  3, 5, 8, ...]
  1 + 1 = 2
   2 + 1 =  3
     3 + 2 =  5
      5 + 3 =  8 ... 

function fibonacci(length){
    if(length <= 1){
        return [1];
    }
    let fib = [1,1];
    while(fib.length < length){
        let last = fib[fib.length - 1];
        let secondLast = fib[fib.length - 2];
        let current = last + secondLast;
        fib.push(current);
    }
    return fib
}
console.log(fibonacci(6)) [ 1, 1, 2, 3, 5, 8 ]
console.log(fibonacci(3)) [ 1, 1, 2 ]
console.log(fibonacci(12)) 
[
1, 1, 2, 3,
