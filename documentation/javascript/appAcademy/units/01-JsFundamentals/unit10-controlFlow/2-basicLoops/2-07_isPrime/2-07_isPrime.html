7. isPrime()

Returns true if the number is prime false if not. 

function isPrime(num){
    if(num <= 2){
        return false;
    }
    for(let i = 2; i < num; i++){
        if(num % i === 0){
            return false

        }
    }
    return true
}
console.log(isPrime(5)) // true
console.log(isPrime(8)) // false
