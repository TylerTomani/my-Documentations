4. fizzBuzz()

A functions that prints of the numbers between 0 and max that are divisible by 3 or 5 but not both. 

function fizzBuzz(max){
    let nums = " ";
    for(let i = 0; i < max; i++){
        if( (i % 3 === 0 || i % 5 === 0) && !(i % 3 === 0 && i % 5 === 0)){
            nums += i + " ";
        }
    }
    console.log(nums)
}
fizzBuzz(20);// 3 5 6 9 10 12 18
