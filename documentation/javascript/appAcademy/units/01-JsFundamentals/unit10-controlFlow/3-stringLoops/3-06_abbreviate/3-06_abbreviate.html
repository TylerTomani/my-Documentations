6. abbreviate()

 Remove all the vowels from the given string 

const vowels = 'aeiou';
function abbreviate(str){
    let newStr = '';
    for(let i = 0; i <str.length; i++){
        if(!(vowels.includes(str[i]))){
            newStr += str[i];
        }
    }
    return newStr;
}
console.log(abbreviate("wonderful")) // "wndrfl"
console.log(abbreviate("accordian")) // "ccrdn"
