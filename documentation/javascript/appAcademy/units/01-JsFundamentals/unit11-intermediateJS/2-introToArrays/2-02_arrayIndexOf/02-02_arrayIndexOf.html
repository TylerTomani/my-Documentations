02-01-Array indexOf()

Return a boolean; true if the el is in the array and false if not. Use the concept of -1 

function arrayHasEl( arr, el ){
    if(arr.indexOf(el) != -1 ){
        return true
    }
    return false
}
console.log(arrayHasEl([1,5,20,40], 20)) // true
console.log(arrayHasEl(["cat","mouse","dog"], "giraffe")) // false
