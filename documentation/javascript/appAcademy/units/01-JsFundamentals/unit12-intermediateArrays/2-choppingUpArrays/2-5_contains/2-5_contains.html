5. contains()

 The function should returns a boolean indicating whether the targetWord is found inside of the sentence.

Solved without using String's indexOf() or includes() methods.

Important Note: convert strings to arrays to use for and forEach loops on them. && For Some Reason, a forEach() does NOT work on this example 

function containsWord(str,target){
    let strArr = str.split (" ");
    for(let i = 0; i < strArr.length ; i++){
        let wrd = strArr[i];
        if( wrd.toLowerCase() === target.toLowerCase()){
            return true
        }
    }
    return false
}

console.log(containsWord('sounds like a plan', 'like')); 
// true
console.log(containsWord('They are great', 'they')); 
// true
console.log(containsWord('caterpillars are great animals', 'cat')); 
// false
console.log(containsWord('Cast the net', 'internet')); 
// false
