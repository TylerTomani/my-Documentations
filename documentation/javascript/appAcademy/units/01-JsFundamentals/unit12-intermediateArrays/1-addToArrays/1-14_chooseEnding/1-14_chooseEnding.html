14. chooseEnding()

 The functions accepts an array of words and a suffix, then returns an array of the words that End with that suffix

 The built in funcion endWith() will return the element if it ends with the provided characters 

function chooseEnding(arr, suffix){
    let newArr =[];
    for(let i = 0; i < arr.length; i++){
        let el  = arr[i];
        if(el.endsWith(suffix)){
            newArr.push(el)
        }
    }
    return newArr;
}
console.log(chooseEnding(['family', 'hound', 'catalyst','fly', 'timidly', 'bond'], 'ly'));
// [ 'family', 'fly', 'timidly' ]

console.log(chooseEnding(['family', 'hound', 'catalyst','fly', 'timidly', 'bond'], 'nd'));
// [ 'hound', 'bond' ]

console.log(chooseEnding(['simplicity', 'computer', 'felicity'], 'icity'));
// [ 'simplicity', 'felicity' ]

console.log(chooseEnding(['simplicity', 'computer', 'felicity'], 'ily'));
// [ ]
