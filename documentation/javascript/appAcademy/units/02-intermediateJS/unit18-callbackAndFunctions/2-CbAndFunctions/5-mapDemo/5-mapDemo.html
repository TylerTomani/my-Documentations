myMap()

 The following code makes the map method using a callback; 

            let people = ['aaron', 'bobby', 'joe', 'tony'];

            let myMap = function(array, cb){
                let newArray = [];
                for( let i = 0; i < array.length; i++){
                    let el = array[i];
                    let capitalPerson = cb(el);
                    newArray.push(capitalPerson);
                }
                return newArray;
            }
            let capitalize = function(el){
                return el.toUpperCase();
            }
            let capitalizedPeople = myMap(people,capitalize)
            console.log(capitalizedPeople)

 Now we can pass in any array and any callback function to perform an action on the array 

        let myMap = function(array,cb){
            let newArr = [];
            for(let i = 0; i <array.length; i++){
                let el = array[i];
                let result = cb(el);
                newArr.push(result);
            }
            return newArr;
        }
        let doubler = myMap([1,4,7,10],function(num){
            return num * 2;
        })
        console.log(doubler)

        let sRoot = myMap([36,81,100,4], Math.sqrt)
        console.log(sRoot)
