 This is an example of using a simple callback function, remember to only pass in the
the name of a callback function as a parameter. Calling a function is
No different then calling what that function returns 

        let superAdd = function(num1,num2,cb){
            sum = num1 + num2;
            let example = cb(sum);
            return example;
        }
        let addHundred = function(num){
            return num + 100;
        }
        console.log(superAdd(8 ,8 , addHundred ))
        // prints 116

 Callback functions are powerful because we can pass different functions into the same parameters 

        let superAdd = function(num1,num2,cb){
            sum = num1 + num2;
            let example = cb(sum);
            return example;
        }
        let addHundred = function(num){
            return num + 100;
        }
        let negate = function(num){
            return -1 * num;
        }
        console.log(superAdd(8 ,8 , addHundred ))
        console.log(superAdd(8 ,8 , negate ))
        // prints 116
        -16

 Anonymous functions can be written as arguments within the parameters. 

        let superAdd = function(num1,num2,cb){
