3-Multiple callbacks

 Below is an example of using two call backs in a function.
The functions returns the callback that produces the larger value. 

    let greaterResult = function(num , cb1, cb2){
        let dbl= cb1(num)
        let squared= cb2(num)
        if( dbl > squared){
            return "double: " + dbl;
        }
        return "squared:" + squared;
    }
    let doubler = function(n){
        return n * 2;
    }
    let squarer = function(n){
        return n * n;
    }
    console.log(greaterResult(2,doubler,squarer))
    console.log(greaterResult(1,doubler,squarer))
    // prints squared:4
              double: 2
