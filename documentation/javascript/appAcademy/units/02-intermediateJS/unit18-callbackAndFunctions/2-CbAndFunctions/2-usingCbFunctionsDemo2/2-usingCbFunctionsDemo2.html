 To change the behavior to make the callback optional add at if statement
to the beggining of the function, 

        let addSroot = function(num1 , num2, cb){
            let sum = num1 + num2;
            if(cb === undefined){
                return sum;
            }

            let sRoot = cb(sum)
            return sRoot;
        }
        console.log(addSroot(30,6))
        console.log(addSroot(30,6, Math.sqrt))
        // prints 36
                  6
