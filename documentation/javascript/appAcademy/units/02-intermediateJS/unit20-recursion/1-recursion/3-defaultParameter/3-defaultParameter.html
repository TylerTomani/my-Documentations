 If we declare a default value for an arguement, we can use it without passing anything in.

let growNumber = function(num, amount = 1){
return num + amount
}
console.log(growNumber(2))
//prints 3

 Default parameters can also be used to pass in flag. This verbose flag will print out each step of the for loop if set to true

let sums = function(nums, verbose = false){
 let total = 0;
 for(let i = 0; i <nums.length; i++){
  if(verbose){
   console.log(`${total} + ${nums[i]} = ${total + nums[i]}`);
  }
  total += num[i];
 }
 return total;
}
let numArr = [1,2,3];
console.log(sums(numArr))
console.log(sums(numArr,true))
// prints 6
// prints 0 + 1 = 1
// 1 + 2 = 3
// 3 + 3 = 6
// 6
