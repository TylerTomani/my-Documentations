6. Rest and Spread

  • The rest parameter syntax accepts an arbitrary number of arguments into a function
  • The spread operator puts multiple objects arrays into one

Rest

function colorPicker(color){
 let string = "I picked the following colors: " + color;
 return string;
}
console.log(colorPicker('red'))
/// returns I picked the following colors: red

 In the above code snipped the function has one parameter color. If we pass in another color, or parameter, nothing will change and the funcion will ignore the second parameter blue

function colorPicker(color){
  let string = "I picked the following colors: " + color;
  return string;
 }
console.log(colorPicker("red","blue"));
/// returns I picked the following colors: red

 Rest allows the use of any amount of arguments within the parameters. We do this by using the ... syntax. This will create an array filled with any extra aguments

function colorPicker(color, ...otherColors ){
  let string = "I picked the following colors: " + color;
  console.log(otherColors)
}
console.log(colorPicker("red","blue","green"));
/// returns ["blue","green"]

  Use a for loop or forEach to iterate through the rest parameters.

function colorPicker(color, ...otherColors){
 let string = " I picked the colors : " + color;

 otherColors.forEach( arg =>{
  string = string + "," + arg
 })
 return string;
}
console.log(colorPicker("red","blue","green","yellow"))
/// returns I picked the following colors: red,blue,green,yellow

 We could also just use ... rest to use all the parameters

function useArgs(...rest){
 rest.forEach(arg => {
 console.log(arg)
 })
}
useArgs("Orange", 22, "Larry")
// prints Orange 22 Larry

 Which is great for compounding numbers for a total sum

function totalSum(...args){
   let sum = 0;
   args.forEach(arg => {
    sum += arg;
   })
   return sum
}
console.log(totalSum(1,5,8,6))
// prints 20

Spread

  Spread is similar to the concat method which combines arrays

let array3 = array1.concat(array2)

 but spread also works with objects


let array1 = ['zebra','bird','parrot'];
let array2 = ['lion','bison','panther'];
let array3 = [...array1, ...array2 ]
console.log(array3)
//prints [ 'zebra', 'bird', 'parrot', 'lion', 'bison', 'panther' ]

let obj1 = { car: "Nissan", color: "blue"};
let obj2 = { wheels: 4, speakers: "bose"};
let obj3 = {...obj1, ...obj2 };
console.log(obj3)
//prints { car: 'Nissan', color: 'blue', wheels: 4, speakers: 'bose' }

Spread as a parameter

 We can spread an object as an arguements into a functions parameters 

        function speak(verb, noun){
            console.log("I like to " + verb + " with " + noun)
        }
        const words = ['run', 'Rob'];
        speak(...words)
        //prints I like to run with Rob

Important Note

 Always Keep in mind that a function will still run and return undefined even if no arguments have been passed in.

function tester(arg) {
 return arg;
}

console.log(tester(5)); // => prints: 5
console.log(tester()); // => prints: undefined
